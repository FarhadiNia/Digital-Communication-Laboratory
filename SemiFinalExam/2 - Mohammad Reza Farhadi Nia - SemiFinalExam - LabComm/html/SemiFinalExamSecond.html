RZ duty = 0.25

b = Binary_Random_Input; l=length(b); b(l+1)=0; n=1; duty = 0.25;
Collect = [];

while n<=l
    t=(n-1):.001:n;
    if b(n)==1
        y = 10*(t<=n-(1-duty) & t>=(n-1))-5;
        Collect = [Collect y];
    elseif b(n)==0
        y = -10*(~(t>=n-(1-duty) & t<=(n-1)))+5;
        Collect = [Collect y];
    else
        y=0;
        Collect = [Collect y];
    end
    subplot(4,1,3);plot(t,y)
    hold on; grid on;
    axis([0 length(b) -10 10]);
    n=n+1;
end
